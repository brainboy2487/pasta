# SECTION 25: RET.NOW inside WHILE (accumulator pattern)
# ════════════════════════════════════════════════════════════════
PRINT "=== SECTION 25: RET.NOW + WHILE combo ==="
DEF count_down(start):
    i = start
    acc = 0
    WHILE i > 0:
        acc = acc + i
        i = i - 1
    END
END
PRINT count_down(5)          # => 15  (5+4+3+2+1)
PRINT count_down(10)         # => 55  (10+9+...+1)

DEF first_over(threshold):
   i = 0
    WHILE i < 100:
        i = i + 1
        IF i * i > threshold:
            RET.NOW(): i
        END
    END
    RET.NOW(): -1
END
PRINT first_over(10)         # => 4  (4*4=16 > 10)
PRINT first_over(100)        # => 11 (11*11=121 > 100)

DEF is_prime(n):
    IF n < 2:
        RET.NOW(): false
    END
    i = 2
    WHILE i * i <= n:
        IF n % i == 0:
            RET.NOW(): false
        END
        i = i + 1
    END
    RET.NOW(): true
END

PRINT is_prime(1)            # => false
PRINT is_prime(2)            # => true
PRINT is_prime(7)            # => true
PRINT is_prime(9)            # => false
PRINT is_prime(13)           # => true
PRINT is_prime(100)          # => false

DEF nth_fib_gt(threshold):
    a = 0
    b = 1
    WHILE b <= threshold:
        temp = b
        b = a + b
        a = temp
    END
    RET.NOW(): b
END
PRINT nth_fib_gt(10)         # => 13
PRINT nth_fib_gt(50)         # => 55
PRINT nth_fib_gt(100)        # => 144

DEF collatz_steps(n):
    steps = 0
    WHILE n != 1:
        IF n % 2 == 0:
            n = n / 2
        OTHERWISE:
            n = n * 3 + 1
        END
        steps = steps + 1
    END
    RET.NOW(): steps
END
PRINT collatz_steps(6)       # => 8
PRINT collatz_steps(27)      # => 111
